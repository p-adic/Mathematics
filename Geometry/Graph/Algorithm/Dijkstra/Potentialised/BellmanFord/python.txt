#有向グラフに対する実装。
#E[始点番号]は辺番号の配列。
#Fは[始点番号,終点番号,重み]の配列。
#B[辺番号]は辺が貼られていることの真偽。
#O(NM+Q(N+M)log N)でWにsからの最短経路長をQ回格納する。
N,M=J()
E=[[]for i in R(N)]
F=[]
for m in R(M):u,v,w=J();E[u-1]+=[m];F+=[[u-1,v-1,w]]
P=[9**20]*N
P[s]=0
for _ in R(N):
	for i,j,w in F:P[j]=min(P[j],P[i]+w)
import heapq
for q in R(Q):
	#ここでクエリによるBの更新を行う。
	
	W=[9**20]*N
	W[s]=0
	S=[[0,s]]
	while S:
		w,i=heapq.heappop(S)
		if w>W[i]:continue
		for e in E[i]:
			if B[e]:
				i,j,v=F[e];v+=P[i]-P[j]
				if w+v<W[j]:W[j]=w+v;heapq.heappush(S,[w+v,j])
	for i in R(N):W[i]+=P[i]
