#Xの中身をソートせずに直接圧縮して種類数を返す。
def UnsortedCoordinateCompress(X):
	Z={}
	for i in R(len(X)):
		if X[i] in Z:X[i]=Z[X[i]]
		else:Z[X[i]]=X[i]=len(Z)
	return len(Z)

#Aの各成分の中身をまとめてソートせずに直接圧縮して種類数を返す。
def MultipleUnsortedCoordinateCompress(A):
	Z={}
	for X in A:
		for i in R(len(X)):
			if X[i] in Z:X[i]=Z[X[i]]
			else:Z[X[i]]=X[i]=len(Z)
	return len(Z)
