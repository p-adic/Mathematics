def B(i,u,F):
	i+=1
	while i<=N:F[i],i=F[i]+u,i+(i&-i)
def M(r,F):
	a=0
	i=min(r+1,N)
	while i:a,i=a+F[i],i-(i&-i)
	return a

def P(l,r,u):
